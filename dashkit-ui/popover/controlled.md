```js
import { Popover, Button } from 'dashkit-ui';

class App extends React.Component {
  state = {
    visible: false,
  }

  render() {
    return (
      <Popover
        content={<a onClick={this.hide} style={{cursor: 'pointer'}}>Close</a>}
        title="Title"
        trigger="click"
        visible={this.state.visible}
      >
        <Button onClick={this.handleClick}>Click me</Button>
      </Popover>
    );
  }

  handleClick = () => {
    const { visible } = this.state;
    this.setState({
      visible: !visible
    });
  }

  hide = () => {
    this.setState({
      visible: false,
    });
  }
}

ReactDOM.render(
  <App />,
  mountNode
);
```
